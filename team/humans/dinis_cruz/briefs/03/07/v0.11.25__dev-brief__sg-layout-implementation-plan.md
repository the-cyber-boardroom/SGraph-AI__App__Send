## 2. sg-layout-events.js (write this first, never change it)

```js
/**
 * sg-layout-events.js
 * Event name constants for the sg-layout framework.
 * Import this file. Never hardcode event name strings.
 * v0.1.0 — these names are frozen. Adding is ok. Changing is a breaking change.
 */

export const SGL_EVENTS = Object.freeze({

    // --- Registration (synchronous, composed: true) ---
    REGISTER:           'sg-layout:register',        // hosted → nearest sg-layout

    // --- Internal bus events (within one sg-layout instance) ---
    PANEL_RESIZED:      'panel:resized',             // { id, width, height }
    PANEL_FOCUSED:      'panel:focused',             // { id }
    PANEL_HIDDEN:       'panel:hidden',              // { id }
    PANEL_SHOWN:        'panel:shown',               // { id }
    PANEL_CLOSED:       'panel:closed',              // { id }
    TAB_CHANGED:        'tab:changed',               // { stackId, tabId }
    LAYOUT_READY:       'layout:ready',              // {} — all panels registered
    LAYOUT_CHANGED:     'layout:changed',            // { tree } — after any structural change

    // --- Upward bubble (composed: true, for cross-boundary drag) ---
    DRAG_START:         'sg-layout:drag-start',      // { panelId, sourceLayoutId }
    DRAG_END:           'sg-layout:drag-end',        // { panelId, accepted: bool }

    // --- Panel API events (dispatched on the sg-layout element itself) ---
    CMD_ADD_PANEL:      'sg-layout:add-panel',       // { tag, title, icon, state, target }
    CMD_REMOVE_PANEL:   'sg-layout:remove-panel',    // { id }
    CMD_FOCUS_PANEL:    'sg-layout:focus-panel',     // { id }
    CMD_SET_LAYOUT:     'sg-layout:set-layout',      // { tree }

    // --- Lifecycle signals (hosted component → parent layout, composed) ---
    SET_TITLE:          'sg-layout:set-title',       // { panelId, title }
    REQUEST_FOCUS:      'sg-layout:request-focus',   // { panelId }
    REQUEST_CLOSE:      'sg-layout:request-close',   // { panelId }
});
```

This file never changes after v0.1.0. New events go in new constants. Old names are never removed — they are deprecated with a comment.

### 3.2 The Layout Tree Implementation

```js
// Internal node constructors — not exported, not custom elements
function makeRow(children, sizes) {
    return { type: 'row', id: uid(), children, sizes: normaliseSizes(sizes, children.length) };
}
function makeColumn(children, sizes) {
    return { type: 'column', id: uid(), children, sizes: normaliseSizes(sizes, children.length) };
}
function makeStack(tabs, activeTab = 0) {
    return { type: 'stack', id: uid(), tabs, activeTab };
}
function makeTab(tag, title, state = {}) {
    return { type: 'tab', id: uid(), tag, title, state, el: null }; // el set on render
}

function normaliseSizes(sizes, count) {
    if (sizes && sizes.length === count) {
        const sum = sizes.reduce((a, b) => a + b, 0);
        return sizes.map(s => s / sum);
    }
    return Array(count).fill(1 / count);
}

function uid() {
    return Math.random().toString(36).slice(2, 10);
}
```

### 3.3 DOM Projection

The tree maps to CSS Grid. Each row/column node becomes a grid container:

```js
_renderNode(node, container) {
    if (node.type === 'row') {
        container.style.display = 'grid';
        container.style.gridTemplateColumns =
            node.children.map((_, i) => {
                const collapsed = this._isCollapsed(node.children[i].id);
                return collapsed ? 'var(--sgl-tab-height)' : `${node.sizes[i]}fr`;
            }).join(` var(--sgl-handle-size) `);
        node.children.forEach((child, i) => {
            const el = this._getOrCreateNodeEl(child);
            container.appendChild(el);
            if (i < node.children.length - 1) {
                container.appendChild(this._makeResizeHandle('col', node, i));
            }
            this._renderNode(child, el);
        });
    }
    // ... column and stack cases similar
}
```

Key rule: **never query the DOM to understand the layout**. Always query `this._tree`. The DOM is write-only from the layout's perspective.

### 3.4 The Registration Handler (the most important method)

```js
_setupRegistrationListener() {
    this._registerDepth = 0;
    this._registeredElements = new Set();

    this.shadowRoot.addEventListener(SGL_EVENTS.REGISTER, (event) => {
        // Guard 1: only handle events from our own slots (not from a nested sg-layout's slots)
        // A nested sg-layout will stopPropagation before this point if it caught it first.
        // We only reach here if this is the nearest sg-layout ancestor.

        event.stopPropagation();

        // Guard 2: cycle detection by call stack depth
        this._registerDepth++;
        if (this._registerDepth > 10) {
            console.error(
                `[sg-layout] Circular registration detected. ` +
                `Element: <${event.target.tagName.toLowerCase()}>. ` +
                `Stack depth: ${this._registerDepth}. Breaking cycle.`
            );
            this._registerDepth--;
            // do not call respond() — component will detect null config and go standalone
            return;
        }

        // Guard 3: cycle detection by element identity
        if (this._registeredElements.has(event.target)) {
            console.warn(
                `[sg-layout] Element <${event.target.tagName.toLowerCase()}> ` +
                `registered twice. Possible cycle. Ignoring second registration.`
            );
            this._registerDepth--;
            return;
        }
        this._registeredElements.add(event.target);

        // Resolve which panel slot this element is in
        const tab = this._findTabForElement(event.target);
        if (!tab) {
            // Element is in our DOM but not in a known panel slot.
            // This is a framework bug — log and fail gracefully.
            console.warn(`[sg-layout] Could not resolve panel for registered element.`);
            this._registerDepth--;
            return;
        }

        // Synchronous respond — this runs before dispatchEvent() returns in the component
        event.detail.respond({
            panelId:   tab.id,
            layoutRef: this,
            depth:     this._depth,         // how many sg-layouts deep we are
            bounds:    this._getTabBounds(tab),
        });

        this._registerDepth--;
    });
}
```

### 3.5 Pre-inject Before connectedCallback

The framework must inject `_sgLayoutRef` and `dataset.panelId` **before** the element's `connectedCallback` fires. This is guaranteed by setting the properties before appending to the DOM:

```js
_mountTab(tab) {
    const el = document.createElement(tab.tag);

    // Inject before connectedCallback fires
    el._sgLayoutRef     = this;
    el.dataset.panelId  = tab.id;
    el.dataset.layoutDepth = String(this._depth);

    if (tab.title) el.dataset.panelTitle = tab.title;
    if (tab.state && typeof el.setLayoutState === 'function') {
        el.setLayoutState(tab.state);   // also before connectedCallback
    }

    tab.el = el;

    // NOW append — connectedCallback fires here
    const slot = this.shadowRoot.querySelector(`slot[name="p-${tab.id}"]`);
    slot.parentElement.appendChild(el);   // el goes to light DOM, projected via slot

    // After connectedCallback: read back any title the component may have set
    if (el.dataset.panelTitle && el.dataset.panelTitle !== tab.title) {
        tab.title = el.dataset.panelTitle;
        this._updateTabChrome(tab);
    }
}
```

### 3.6 Bootstrap Sequence (the full flow)

```
sg-layout element appended to page DOM
│
├─ connectedCallback()
│   ├─ _setupRegistrationListener()    ← shadow listener in place
│   ├─ _parseInitialLayout()           ← build tree from JSON attr or child elements
│   ├─ _hideLayout()                   ← opacity:0 on root container
│   ├─ _renderTree()                   ← project tree to shadow DOM, create slots
│   ├─ _mountAllTabs()                 ← for each tab in tree:
│   │   ├─ inject _sgLayoutRef, dataset.panelId   (before connectedCallback)
│   │   ├─ appendChild to light DOM               (connectedCallback fires here)
│   │   │   └─ component dispatches sg-layout:register
│   │   │       └─ _setupRegistrationListener handler fires synchronously
│   │   │           └─ respond() called with { panelId, layoutRef, depth, bounds }
│   │   │               └─ component stores config, renders itself
│   │   │                   └─ dispatchEvent returns
│   │   └─ [next tab]
│   ├─ events.emit(LAYOUT_READY, {})   ← all tabs mounted and registered
│   └─ requestAnimationFrame(() => _showLayout())  ← reveal after one frame
│
user sees fully rendered layout for the first time
```

### 3.7 Serialisation

```js
getLayout() {
    return JSON.parse(JSON.stringify(this._serializeNode(this._tree)));
}

_serializeNode(node) {
    if (node.type === 'tab') {
        const state = (typeof node.el?.getLayoutState === 'function')
            ? node.el.getLayoutState()
            : node.state;
        return { type: 'tab', id: node.id, title: node.title, tag: node.tag, state };
    }
    if (node.type === 'stack') {
        return { type: 'stack', id: node.id, activeTab: node.activeTab,
                 tabs: node.tabs.map(t => this._serializeNode(t)) };
    }
    return { type: node.type, id: node.id, sizes: [...node.sizes],
             children: node.children.map(c => this._serializeNode(c)) };
}

setLayout(json) {
    this._unmountAll();
    this._tree = this._deserializeNode(json);
    this._renderTree();
    this._mountAllTabs();
}
```

### 3.8 Resize Handle Implementation

Resize handles use `pointermove` (not `mousemove`) — works for both mouse and touch:

```js
_makeResizeHandle(axis, parentNode, index) {
    const handle = document.createElement('div');
    handle.className = `sgl-resize-handle sgl-resize-handle--${axis}`;

    let start, startSizes;

    handle.addEventListener('pointerdown', e => {
        e.preventDefault();
        handle.setPointerCapture(e.pointerId);
        start = axis === 'col' ? e.clientX : e.clientY;
        startSizes = [...parentNode.sizes];
        handle.classList.add('sgl-resize-handle--active');
    });

    handle.addEventListener('pointermove', e => {
        if (!handle.hasPointerCapture(e.pointerId)) return;

        const containerSize = axis === 'col'
            ? this.shadowRoot.querySelector('.sgl-root').offsetWidth
            : this.shadowRoot.querySelector('.sgl-root').offsetHeight;

        const delta = (axis === 'col' ? e.clientX : e.clientY) - start;
        const deltaFraction = delta / containerSize;

        const newSizes = [...startSizes];
        const minFraction = 80 / containerSize;

        newSizes[index]     = Math.max(minFraction, startSizes[index] + deltaFraction);
        newSizes[index + 1] = Math.max(minFraction, startSizes[index + 1] - deltaFraction);

        // Renormalise so they still sum to startSizes[index] + startSizes[index+1]
        const total = startSizes[index] + startSizes[index + 1];
        const newTotal = newSizes[index] + newSizes[index + 1];
        newSizes[index]     = newSizes[index] / newTotal * total;
        newSizes[index + 1] = newSizes[index + 1] / newTotal * total;

        parentNode.sizes = newSizes;
        this._applyGridSizes(parentNode);    // update CSS only, no full re-render

        // Notify affected panels
        this._notifyPanelResizes(parentNode);
    });

    handle.addEventListener('pointerup', e => {
        handle.releasePointerCapture(e.pointerId);
        handle.classList.remove('sgl-resize-handle--active');
        this._events.emit(SGL_EVENTS.LAYOUT_CHANGED, { tree: this.getLayout() });
    });

    return handle;
}

_notifyPanelResizes(node) {
    // Walk the subtree to find all leaf tabs and emit resize events
    this._walkTabs(node, tab => {
        if (tab.el) {
            const bounds = this._getTabBounds(tab);
            this._events.emit(`panel:${tab.id}:resized`, bounds);
        }
    });
}
```

## 4. v0.1.1 — Tab Stacks

### Scope

- Multiple hosted elements in one stack, with a tab bar
- Click tab to switch active panel (hide/show via CSS, not DOM remove/add)
- Close tab button per tab
- `TAB_CHANGED` event on internal bus
- Drag tab within the same stack to reorder (no cross-stack drag yet — that's v0.1.3)

### Key implementation note: hide, don't remove

When switching tabs, the inactive elements are hidden with `display:none` (or `visibility:hidden` + `height:0` to preserve scroll position). They are **never removed from the DOM**. Removing and re-appending would re-trigger `connectedCallback` — the component would go through registration again and potentially re-render.

```js
_activateTab(stack, tabIndex) {
    stack.tabs.forEach((tab, i) => {
        if (!tab.el) return;
        tab.el.style.display = i === tabIndex ? '' : 'none';
        this._events.emit(
            i === tabIndex ? SGL_EVENTS.PANEL_SHOWN : SGL_EVENTS.PANEL_HIDDEN,
            { id: tab.id }
        );
    });
    stack.activeTab = tabIndex;
    this._updateTabBarChrome(stack);
    this._events.emit(SGL_EVENTS.TAB_CHANGED, { stackId: stack.id, tabId: stack.tabs[tabIndex].id });
}
```

## 5. v0.1.2 — Fractal Registration

### Scope

This version ensures the fractal case works correctly end-to-end with the full registration chain and `depth` tracking. The core mechanism already works in v0.1.0 (the registration listener uses `stopPropagation`), but this version adds:

- `depth` correctly propagated from outer to inner `sg-layout`
- Inner `sg-layout` stores `_parentLayoutRef` for the cross-boundary drag case (v0.1.4)
- Serialisation preserves inner layout state in the `state` field of the parent tab
- Restore correctly rebuilds inner `sg-layout` with its own tree

### The one tricky case: inner sg-layout as tab

When the outer layout adds an `sg-layout` as a tab, the inner layout's `connectedCallback` fires and it registers with the outer layout. The outer layout calls `respond()` with `depth: 1`. The inner layout stores this.

Then the inner layout builds its own tree and mounts its own tabs. Each of those tabs dispatches `sg-layout:register`. **The inner layout's `_setupRegistrationListener` catches these first** (because `stopPropagation` in the inner layout's handler prevents them reaching the outer layout). Each inner tab gets `depth: 2`.

This chain extends arbitrarily. At each level, `stopPropagation` ensures the event is handled by the nearest ancestor.

## 6. v0.1.3 — Drag to Dock

### Scope

- Drag a panel header within a single `sg-layout` instance
- Drop zones appear at panel edges and centre (split left/right/top/bottom/tab)
- Drop restructures the layout tree
- Ghost element follows cursor during drag
- No cross-layout drag yet (v0.1.4)

### Implementation strategy

This is the most complex minor version. Isolate the drag system as a separate internal class `LayoutDragController` that the main `SgLayout` class instantiates. This keeps the main class readable.

```js
// internal — not exported
class LayoutDragController {
    constructor(layout) {
        this._layout = layout;
        this._dragging = null;      // { tab, sourceStack, ghost }
        this._dropTarget = null;    // { node, position: 'left'|'right'|'top'|'bottom'|'tab' }
    }

    start(tab, sourceStack, pointerEvent) { ... }
    move(pointerEvent) { ... }     // update ghost position + highlight drop zones
    end(pointerEvent) { ... }      // commit tree restructure or cancel

    _computeDropTarget(x, y) { ... }     // hit test against panel rects
    _showDropZones() { ... }             // add overlay elements to shadow DOM
    _hideDropZones() { ... }             // remove overlay elements
    _restructureTree(tab, source, target, position) { ... }  // tree mutation
}
```

Drop zone hit testing uses element bounding rects. No custom hit-test geometry — just `getBoundingClientRect()` on each panel slot wrapper.

## 7. v0.1.4 — Cross-Layout Drag

### Scope

Drag a panel from an inner `sg-layout` to an outer `sg-layout` (or vice versa). This requires the upward bubble channel.

The mechanism:
1. Inner layout's `LayoutDragController` detects pointer leaving its bounds
2. Inner layout dispatches `sg-layout:drag-start` with `composed:true` — the drag event bubbles up through shadow boundaries
3. Outer layout catches it, activates its own drop zones
4. On drop: outer layout calls `innerLayout.removePanel(id)` then adds the element to its own tree
5. The hosted element is re-slotted — it fires `sg-layout:register` again with the outer layout as the new parent
6. Outer layout responds with new `panelId`, `bounds`
7. Element updates its `_sgLayoutRef` and `dataset.panelId`

The component experiences this as: `panel:hidden` → brief gap → `panel:shown` with new dimensions. It does not observe the reparenting.

## 8. v0.1.5 — Floating Panels

### Scope

- Detach a panel from the layout tree
- Panel becomes `position: fixed`, draggable anywhere on screen
- Re-dock by dragging to a drop zone in any `sg-layout`

Floating panels live outside the layout tree. They are rendered in a separate `<div>` appended to `document.body` (not inside any `sg-layout` shadow root). The hosted element moves with them. `_sgLayoutRef` is cleared. If the component uses it, it gets null and handles accordingly.

## 11. sg-commands Module — Implementation Plan

**Location:** `core/sg-commands/v0.1.0/sg-commands.js`
**Dependency on sg-layout:** None. Companion module, completely independent.
**Depends on:** `sg-panel-identity` (optional — for encrypted mode)

### The module

```js
// core/sg-commands/v0.1.0/sg-commands.js

export class SgCommandBus {
    constructor() {
        this._subscribers = new Map();   // panelId → handler function
        this._layout      = null;        // sg-layout ref for path resolution (optional)
    }

    // Called by sg-layout after mounting panels
    attachLayout(layoutEl) {
        this._layout = layoutEl;
    }

    // Component calls this in connectedCallback
    subscribe(panelId, handler) {
        this._subscribers.set(panelId, handler);
    }

    unsubscribe(panelId) {
        this._subscribers.delete(panelId);
    }

    // Send to one panel by id
    send(panelId, command) {
        const handler = this._subscribers.get(panelId);
        if (handler) handler(command);
    }

    // Send to all panels matching a path expression
    sendTo(path, command) {
        if (!this._layout) return;
        const ids = this._layout.resolvePath(path);   // returns array of panelIds
        ids.forEach(id => this.send(id, command));
    }

    // Send to all subscribed panels
    broadcast(command) {
        for (const handler of this._subscribers.values()) {
            try { handler(command); } catch (e) { console.error('[sg-commands]', e); }
        }
    }
}

// Page-level singleton — optional, apps can instantiate their own
export const sgCommands = new SgCommandBus();

// Auto-attach to any sg-layout on the page
document.addEventListener('DOMContentLoaded', () => {
    const layouts = document.querySelectorAll('sg-layout');
    layouts.forEach(l => sgCommands.attachLayout(l));
});
```

### Component contract

Components declare what commands they handle:

```js
class VideoSplitter extends HTMLElement {
    // Declare supported commands — framework uses this for validation/docs
    static sgCommands = ['show', 'hide', 'state-push', 'theme-change'];

    connectedCallback() {
        // ... registration ...

    }

    _handleCommand(cmd) {
        switch (cmd.type) {
            case 'hide':         this.style.display = 'none'; break;
            case 'show':         this.style.display = '';     break;
            case 'state-push':   this._applyState(cmd.payload); break;
            case 'theme-change': this._applyTheme(cmd.payload); break;
            // unknown types: silently ignored
        }
    }
}
```

### The module

```js
export class SgPanelIdentity {
    constructor() {
        this._registry = new Map();   // panelId → { publicKey, privateKey }
    }

    async assignIdentities(layoutEl) {
        const panels = layoutEl.getAllPanels();   // returns array of { id, el }

        // Generate all keypairs in parallel
        const pairs = await Promise.all(panels.map(async panel => ({
            id: panel.id,
            keys: await crypto.subtle.generateKey(
                { name: 'ECDH', namedCurve: 'P-256' },
                true,
                ['deriveKey', 'deriveBits']
            )
        })));

        // Assign to panels
        for (const { id, keys } of pairs) {
            this._registry.set(id, keys);
            const panel = panels.find(p => p.id === id);
            if (panel?.el) {
                panel.el._sgPrivateKey = keys.privateKey;
                // Export public key to dataset for inspection/serialisation
                const exported = await crypto.subtle.exportKey('raw', keys.publicKey);
                panel.el.dataset.panelPublicKey = btoa(
                    String.fromCharCode(...new Uint8Array(exported))
                );
            }
        }
    }

    getPublicKey(panelId) {
        return this._registry.get(panelId)?.publicKey ?? null;
    }

    async exportKeypair(panelId) {
        const pair = this._registry.get(panelId);
        if (!pair) return null;
        const [pub, priv] = await Promise.all([
            crypto.subtle.exportKey('raw', pair.publicKey),
            crypto.subtle.exportKey('pkcs8', pair.privateKey),
        ]);
        return {
            publicKey:  btoa(String.fromCharCode(...new Uint8Array(pub))),
            privateKey: btoa(String.fromCharCode(...new Uint8Array(priv))),
        };
    }

    async importKeypair(panelId, exported) {
        // Restore a keypair from serialised data
        const pubRaw  = Uint8Array.from(atob(exported.publicKey), c => c.charCodeAt(0));
        const privRaw = Uint8Array.from(atob(exported.privateKey), c => c.charCodeAt(0));
        const [pub, priv] = await Promise.all([
            crypto.subtle.importKey('raw',   pubRaw,  { name: 'ECDH', namedCurve: 'P-256' }, true, []),
            crypto.subtle.importKey('pkcs8', privRaw, { name: 'ECDH', namedCurve: 'P-256' }, true, ['deriveKey', 'deriveBits']),
        ]);
        this._registry.set(panelId, { publicKey: pub, privateKey: priv });
    }
}

export const sgPanelIdentity = new SgPanelIdentity();
```

## 13. Digital Twin — Full Serialisation Implementation

Add to `sg-layout`'s public API:

```js
// Returns the complete digital twin — layout tree + all component states + geometry + identity
async getTwin() {
    const tree = await this._serializeTwinNode(this._tree);
    return {
        sgLayoutVersion: 'v0.1.0',
        id:              this._id,
        timestamp:       Date.now(),
        geometry: {
            width:  this.offsetWidth,
            height: this.offsetHeight,
        },
        tree,
    };
}

async _serializeTwinNode(node) {
    if (node.type === 'tab') {
        // Get component state — prefer async, fall back to sync, fall back to {}
        let componentState = {};
        if (typeof node.el?.getLayoutStateAsync === 'function') {
            componentState = await node.el.getLayoutStateAsync();
        } else if (typeof node.el?.getLayoutState === 'function') {
            componentState = node.el.getLayoutState();
        }

        // If this tab hosts an inner sg-layout, componentState IS the inner twin
        if (node.tag === 'sg-layout' && node.el) {
            componentState = await node.el.getTwin();
        }

        // Export public key if identity module is active
        let publicKey = null;
        if (window.sgPanelIdentity) {
            const exported = await window.sgPanelIdentity.exportKeypair(node.id);
            publicKey = exported?.publicKey ?? null;
        }

        return { type: 'tab', id: node.id, tag: node.tag, title: node.title,
                 publicKey, componentState };
    }
    if (node.type === 'stack') {
        return { type: 'stack', id: node.id, activeTab: node.activeTab,
                 tabs: await Promise.all(node.tabs.map(t => this._serializeTwinNode(t))) };
    }
    return { type: node.type, id: node.id, sizes: [...node.sizes],
             children: await Promise.all(node.children.map(c => this._serializeTwinNode(c))) };
}

// Restore from a twin document — including keypairs if present
async setTwin(twin) {
    this._id = twin.id;
    this._unmountAll();
    this._tree = this._deserializeNode(twin.tree);
    this._renderTree();
    await this._mountAllTabsFromTwin(twin.tree);

    // Restore keypairs if identity module is active and twin contains them
    if (window.sgPanelIdentity) {
        await this._restoreIdentities(twin.tree);
    }
}
```

## 14. Mini DSL — Path Resolver + Builder

### Path resolver (add to sg-layout)

```js
// Returns array of panelIds matching a path expression
resolvePath(path) {
    const segments = path.split('/');
    return this._resolveSegments([this._tree], segments);
}

_resolveSegments(nodes, segments) {
    if (segments.length === 0) return [];
    const [head, ...rest] = segments;

    let matched = [];

    if (head === '**') {
        // Deep wildcard — collect all tab ids in subtree
        matched = nodes.flatMap(n => this._collectAllTabIds(n));
    } else if (head === '*') {
        matched = nodes.flatMap(n => n.type === 'tab' ? [n.id] : (n.children ?? n.tabs ?? []).map(c => c.id));
    } else if (head.match(/^(row|column|stack|tab)\[(\d+)\]$/)) {
        const [, type, idx] = head.match(/^(\w+)\[(\d+)\]$/);
        matched = nodes.flatMap(n => {
            const children = n.children ?? n.tabs ?? [];
            const child = children.filter(c => c.type === type)[parseInt(idx)];
            return child ? [child] : [];
        });
    } else {
        // Treat as a literal panelId
        matched = nodes.filter(n => n.id === head);
    }

    if (rest.length === 0) {
        return matched.filter(n => n.type === 'tab').map(n => n.id);
    }
    return this._resolveSegments(matched, rest);
}
```

## 15. Stress Test — 32×32 Pixel Grid

**Location:** `tests/sg-layout/stress-32x32.html`

```html
<!DOCTYPE html>
<html>
<head>
    <script type="module" src="../../core/sg-layout/v0.1.0/sg-layout.js"></script>

    function buildGrid() {
        return {
            type: 'column',
            sizes: Array(ROWS).fill(1/ROWS),
            children: Array.from({ length: ROWS }, (_, r) => ({

                bubbles: true, composed: true,

            }));
            this._panelId = config?.panelId;
```

| Metric | Record | Decision threshold |
|---|---|---|
| Mount time (1024 panels) | ms | > 500ms → investigate |
| Frame colour update time | ms | > 16ms → CSS Grid too slow, switch to `position:absolute` |
| `getTwin()` time | ms | > 200ms → async serialisation needed |
| Twin JSON size | bytes | — (informational) |
| Registration depth reached | count | Should be exactly 2 (column→row→stack) |
