## PKI__Crypto.py — Reference Implementation

This is the core class. Follow the `Vault__Crypto` pattern: `Type_Safe` base class, pure functions, no side effects.

```python
from cryptography.hazmat.primitives.asymmetric     import rsa, ec, padding, utils as asym_utils
from cryptography.hazmat.primitives                 import hashes, serialization
from cryptography.hazmat.primitives.ciphers.aead    import AESGCM
from osbot_utils.type_safe.Type_Safe                import Type_Safe
import hashlib, os, base64, json

class PKI__Crypto(Type_Safe):

    # --- Key Generation ---

    def generate_encryption_key_pair(self):
        """Generate RSA-OAEP 4096-bit key pair for encryption."""
        private_key = rsa.generate_private_key(
            public_exponent = 65537,
            key_size        = 4096)
        return private_key, private_key.public_key()

    def generate_signing_key_pair(self):
        """Generate ECDSA P-256 key pair for signing."""
        private_key = ec.generate_private_key(ec.SECP256R1())
        return private_key, private_key.public_key()

    # --- PEM Export ---

    def export_public_key_pem(self, public_key):
        return public_key.public_bytes(
            serialization.Encoding.PEM,
            serialization.PublicFormat.SubjectPublicKeyInfo
        ).decode()

    def export_private_key_pem(self, private_key, passphrase=None):
        encryption = (serialization.BestAvailableEncryption(passphrase.encode())
                      if passphrase
                      else serialization.NoEncryption())
        return private_key.private_bytes(
            serialization.Encoding.PEM,
            serialization.PrivateFormat.PKCS8,
            encryption
        ).decode()

    # --- PEM Import ---

    def import_public_key_pem(self, pem_str):
        return serialization.load_pem_public_key(pem_str.encode())

    def import_private_key_pem(self, pem_str, passphrase=None):
        pwd = passphrase.encode() if passphrase else None
        return serialization.load_pem_private_key(pem_str.encode(), password=pwd)

    # --- Fingerprint (MUST match browser and server) ---

    def compute_fingerprint(self, public_key):
        der = public_key.public_bytes(
            serialization.Encoding.DER,
            serialization.PublicFormat.SubjectPublicKeyInfo)
        digest = hashlib.sha256(der).hexdigest()
        return f"sha256:{digest[:16]}"

    # --- Signing (ECDSA P-256, raw r||s for Web Crypto interop) ---

    def sign(self, private_key, data):
        """Sign data with ECDSA P-256. Returns raw r||s bytes (64 bytes)."""
        der_sig = private_key.sign(data, ec.ECDSA(hashes.SHA256()))
        r, s = asym_utils.decode_dss_signature(der_sig)
        return r.to_bytes(32, 'big') + s.to_bytes(32, 'big')

    def verify(self, public_key, signature_raw, data):
        """Verify raw r||s signature (64 bytes) from Web Crypto."""
        r = int.from_bytes(signature_raw[:32], 'big')
        s = int.from_bytes(signature_raw[32:], 'big')
        der_sig = asym_utils.encode_dss_signature(r, s)
        public_key.verify(der_sig, data, ec.ECDSA(hashes.SHA256()))
        return True  # raises InvalidSignature on failure

    # --- Hybrid Encryption (RSA-OAEP wraps AES-256-GCM) ---

    def hybrid_encrypt(self, recipient_public_key, plaintext,
                       signing_private_key=None, signing_fingerprint=None):
        """Encrypt plaintext, return v2 payload dict matching browser format."""
        data    = plaintext.encode() if isinstance(plaintext, str) else plaintext
        aes_key = os.urandom(32)
        iv      = os.urandom(12)

        aesgcm     = AESGCM(aes_key)
        ciphertext = aesgcm.encrypt(iv, data, None)

        wrapped_key = recipient_public_key.encrypt(
            aes_key,
            padding.OAEP(
                mgf       = padding.MGF1(algorithm=hashes.SHA256()),
                algorithm = hashes.SHA256(),
                label     = None))

        payload = dict(v = 2,
                       w = base64.b64encode(wrapped_key).decode(),
                       i = base64.b64encode(iv).decode(),
                       c = base64.b64encode(ciphertext).decode())

        if signing_private_key and signing_fingerprint:
            sig_raw    = self.sign(signing_private_key, ciphertext)
            payload['s'] = base64.b64encode(sig_raw).decode()
            payload['f'] = signing_fingerprint

        return base64.b64encode(json.dumps(payload).encode()).decode()

    def hybrid_decrypt(self, recipient_private_key, encoded_payload,
                       contacts_keyring=None):
        """Decrypt v2 payload, optionally verify signature."""
        payload = json.loads(base64.b64decode(encoded_payload))

        if payload.get('v') not in (1, 2):
            raise ValueError(f"Unsupported payload version: {payload.get('v')}")

        wrapped_key = base64.b64decode(payload['w'])
        iv          = base64.b64decode(payload['i'])
        ciphertext  = base64.b64decode(payload['c'])

        aes_key = recipient_private_key.decrypt(
            wrapped_key,
            padding.OAEP(
                mgf       = padding.MGF1(algorithm=hashes.SHA256()),
                algorithm = hashes.SHA256(),
                label     = None))

        aesgcm    = AESGCM(aes_key)
        plaintext = aesgcm.decrypt(iv, ciphertext, None)

        result = dict(plaintext = plaintext.decode(),
                      signed    = False,
                      verified  = False,
                      signer    = None)

        if payload.get('s') and payload.get('f') and contacts_keyring:
            result['signed'] = True
            sig_raw = base64.b64decode(payload['s'])
            contact = contacts_keyring.lookup_by_signing_fingerprint(payload['f'])
            if contact:
                try:
                    self.verify(contact.signing_public_key, sig_raw, ciphertext)
                    result['verified'] = True
                    result['signer']   = contact.label
                except Exception:
                    result['verified'] = False

        return result
```

## Key Interop Pitfalls

### 1. ECDSA Signature Encoding (highest risk)

Web Crypto uses **IEEE P1363** format (raw r‖s, 64 bytes for P-256).
Python `cryptography` uses **DER** format by default.

**You MUST convert.** The `sign()` and `verify()` methods above handle this. Test with cross-platform vectors.

### 2. AES-GCM Tag Handling

Web Crypto's `encrypt()` returns `ciphertext + tag` concatenated (tag is last 16 bytes).
Python's `AESGCM.encrypt()` also returns `ciphertext + tag` concatenated.

**These are compatible.** No conversion needed.

### 3. RSA-OAEP Parameters

Both sides must use identical OAEP parameters:
- Hash: SHA-256
- MGF: MGF1 with SHA-256
- Label: None (empty)

Web Crypto defaults to SHA-1 for MGF if not specified. The admin UI uses `{ name: 'RSA-OAEP' }` which defaults to SHA-256 for the hash (set at key generation time via `hash: 'SHA-256'`), but MGF1 hash follows the key's hash algorithm.

**Ensure the Python side matches exactly.**

### 4. PEM Line Length

PEM standard requires 64-character lines. The admin UI's `exportPublicKeyPEM` does `b64.match(/.{1,64}/g).join('\n')`. Python's `serialization` module handles this correctly by default.

## Test Strategy

### Unit tests (no mocks, no patches)

```python
# tests/unit/crypto/test_PKI__Crypto.py

class Test__PKI__Crypto:

    def test_generate_and_fingerprint(self):
        """Generate key pair, compute fingerprint, verify format."""
        pki = PKI__Crypto()
        priv, pub = pki.generate_encryption_key_pair()
        fp = pki.compute_fingerprint(pub)
        assert fp.startswith('sha256:')
        assert len(fp) == 23  # "sha256:" + 16 hex chars

    def test_sign_verify_roundtrip(self):
        """Generate signing key, sign message, verify."""
        pki = PKI__Crypto()
        priv, pub = pki.generate_signing_key_pair()
        message = b"test message"
        sig = pki.sign(priv, message)
        assert len(sig) == 64  # raw r||s
        assert pki.verify(pub, sig, message)

    def test_hybrid_encrypt_decrypt_roundtrip(self):
        """Generate encryption key, encrypt, decrypt, verify plaintext."""
        pki = PKI__Crypto()
        priv, pub = pki.generate_encryption_key_pair()
        encoded = pki.hybrid_encrypt(pub, "hello world")
        result = pki.hybrid_decrypt(priv, encoded)
        assert result['plaintext'] == "hello world"

    def test_hybrid_encrypt_decrypt_with_signature(self):
        """Encrypt+sign, decrypt+verify, check signer identity."""
        pki = PKI__Crypto()
        enc_priv, enc_pub = pki.generate_encryption_key_pair()
        sig_priv, sig_pub = pki.generate_signing_key_pair()
        fp = pki.compute_fingerprint(sig_pub)
        # ... encrypt with signing, decrypt with keyring contact

    def test_pem_export_import_roundtrip(self):
        """Export key to PEM, import back, fingerprints match."""
        pki = PKI__Crypto()
        priv, pub = pki.generate_encryption_key_pair()
        pem = pki.export_public_key_pem(pub)
        assert '-----BEGIN PUBLIC KEY-----' in pem
        imported = pki.import_public_key_pem(pem)
        assert pki.compute_fingerprint(pub) == pki.compute_fingerprint(imported)

    def test_fingerprint_matches_server_implementation(self):
        """Verify fingerprint computation matches Service__Keys._compute_fingerprint_from_pem."""
        pki = PKI__Crypto()
        _, pub = pki.generate_encryption_key_pair()
        pem = pki.export_public_key_pem(pub)
        # Compute using CLI method
        fp_cli = pki.compute_fingerprint(pub)
        # Compute using server method (inline)
        import base64, hashlib
        lines = pem.strip().split('\n')
        b64_data = ''.join(line for line in lines if not line.startswith('-----'))
        der = base64.b64decode(b64_data)
        digest = hashlib.sha256(der).hexdigest()
        fp_server = f"sha256:{digest[:16]}"
        assert fp_cli == fp_server
```

## CLI Commands (Phase 3)

```
sg-send-cli pki keygen              # Generate encryption + signing key pair
sg-send-cli pki list                 # List local key pairs
sg-send-cli pki export <fingerprint> # Export public key bundle (JSON)
sg-send-cli pki delete <fingerprint> # Delete key pair (with confirmation)
sg-send-cli pki import <file|->      # Import contact's public key
sg-send-cli pki contacts             # List imported contacts
sg-send-cli pki sign <file>          # Sign file, output detached signature
sg-send-cli pki verify <file> <sig>  # Verify detached signature
sg-send-cli pki encrypt <file>       # Encrypt for recipient
sg-send-cli pki decrypt <file>       # Decrypt with local key
sg-send-cli pki publish              # Publish public key to registry
sg-send-cli pki lookup <code>        # Lookup key by registry code
```

## What NOT to Implement

1. **ECDH P-256 encryption** — not in scope, admin UI encrypt/decrypt only uses RSA-OAEP
2. **X.509 certificates** — future work, not needed for messaging
3. **Ed25519** — not compatible with existing admin UI
4. **Key rotation** — future feature, not Phase 3
5. **Hardware key support (FIDO2, TPM)** — future feature
